\begin{frame}{Question}
The time between successive failures (in hours) of a side discharge loader operating in a mechanised underground coal mine are as follows:
\begin{center}
    62, 58, 54, 50, 52, 60, 58, 57, 50, 53
\end{center}
If the failure data follow an exponential distribution, then reliability of the equipment for a period of 50 hours is:
\vspace{30pt}
\begin{columns}[t]
    \column{0.2\textwidth}
    1) 0.25
    \column{0.2\textwidth}
    2) 0.40
    \column{0.2\textwidth}
    3) 0.60
    \column{0.2\textwidth}
    4) 1.00
\end{columns}
\end{frame}


\begin{frame}{ Parameter Estimation}
    For an exponential distribution, we first find the Mean Time Between Failures $(\mu)$:
    \begin{equation*}
       \mu = \frac{\sum t_i}{n} = \frac{554}{10} = 55.4 \text{ hours}
    \end{equation*}
    
    As the given is a exponential distribution, failure rate ($\lambda$) is the reciprocal of the $\mu$:
    \begin{equation*}
        \lambda = \frac{1}{\mu} = \frac{1}{55.4} \approx 0.01805 \text{ failures/hour}
    \end{equation*}
\end{frame}

\begin{frame}{ PDF and CDF Definitions}
    \textbf{Probability Density Function (PDF):}
    Represents the failure density at time $t$.
    \begin{equation*}
        f(t) = \lambda e^{-\lambda t}
    \end{equation*}

    \textbf{Cumulative Distribution Function (CDF):}
    Represents the probability that failure occurs \textit{before} time $t$.
    \begin{equation*}
        F(t) = P(T \le t) = \int_{0}^{t} \lambda e^{-\lambda x} dx = 1 - e^{-\lambda t}
    \end{equation*}
\end{frame}

\begin{frame}{ Reliability Calculation}
    Reliability $R(t)$ is the probability of survival beyond time $t$. It is the complement of the CDF:
    \begin{equation*}
        R(t) = 1 - F(t) = e^{-\lambda t}
    \end{equation*}

    Substitute $t = 50$ and $\lambda = 0.01805$:
    \begin{itemize}
        \item $R(50) = e^{-(0.01805)(50)}$
        \item $R(50) = e^{-0.9025}$
        \item $R(50) \approx 0.4055$
    \end{itemize}
\end{frame}


\begin{frame}{Final Conclusion}
    \begin{block}{Result}
        The reliability of the side discharge loader for a 50-hour period is approximately \textbf{0.40}.
    \end{block}
    
    \vspace{0.5cm}
    \textbf{Correct Option: (b)}
    \vfill
    \begin{itemize}
        \item $P(\text{Failure before 50h}) \approx 60\%$ (CDF)
        \item $P(\text{Survival past 50h}) \approx 40\%$ (Reliability)
    \end{itemize}
\end{frame}

\begin{mycode}
import numpy as np
import matplotlib.pyplot as plt

# Parameters from the problem
data = [62, 58, 54, 50, 52, 60, 58, 57, 50, 53]
mtbf = np.mean(data)
lambd = 1 / mtbf
t_val = 50

# Time range for plotting
t = np.linspace(0, 200, 1000)

# CDF: F(t) = 1 - exp(-lambda * t)
cdf = 1 - np.exp(-lambd * t)




\end{mycode}
